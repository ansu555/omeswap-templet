# Phase 2 — Agent 4: Regime Detection
> **Depends on:** Phase 1 (`price_buffer:BTC`, `price_buffer:ETH` in Redis)
> **Unlocks:** Phase 3 (Signal Agent reads `regime:current` before every cycle)
> **Estimated effort:** 2–3 days

## Goal

Give the entire system its global market context. Agent 4 runs every 15 minutes, reads BTC/ETH price data and the Binance perpetual funding rate, feeds them into a pre-trained Hidden Markov Model, and writes the resulting regime label + confidence to `regime:current` in Redis. Every other agent reads this key first — it controls which indicators Signal Agent trusts and how aggressively Risk Agent sizes positions.

This phase also produces the HMM training script so you can retrain the model on fresh data monthly.

## What gets built

- `ats/agents/agent4_regime.py` — main regime classification loop
- `ats/agents/regime/hmm_model.py` — HMM wrapper (train + predict)
- `ats/agents/regime/feature_builder.py` — assembles the 3-column feature matrix from Redis data
- `ats/agents/regime/funding_rate.py` — Binance futures API client for funding rate
- `scripts/train_regime_model.py` — one-shot training script (run once, save model weights)
- Model weights saved to `models/regime_hmm.pkl`

## Step-by-step implementation

### Step 1 — Funding rate fetcher

```python
# ats/agents/regime/funding_rate.py
import httpx

BINANCE_FUTURES_URL = "https://fapi.binance.com/fapi/v1/fundingRate"
PAIRS = ["BTCUSDT", "ETHUSDT"]

async def get_avg_funding_rate() -> float:
    """Return average perpetual funding rate across BTC and ETH (latest period)."""
    async with httpx.AsyncClient() as client:
        rates = []
        for symbol in PAIRS:
            try:
                resp = await client.get(
                    BINANCE_FUTURES_URL,
                    params={"symbol": symbol, "limit": 1},
                    timeout=5,
                )
                data = resp.json()
                if data:
                    rates.append(float(data[-1]["fundingRate"]))
            except Exception:
                pass
        return sum(rates) / len(rates) if rates else 0.0
```

### Step 2 — Feature builder

Assembles the 3-column matrix from Redis price buffers.

```python
# ats/agents/regime/feature_builder.py
import json
import numpy as np
from ats.data.redis_client import get_redis

async def build_features() -> np.ndarray | None:
    """
    Returns shape (N, 3): [btc_daily_return, btc_30d_realized_vol, funding_rate]
    Returns None if not enough data in Redis.
    """
    r = await get_redis()
    raw = await r.lrange("price_buffer:BTC", 0, 59)
    if len(raw) < 30:
        return None   # need at least 30 bars

    prices = [json.loads(b).get("price", 0) for b in reversed(raw)]
    prices = np.array(prices, dtype=float)

    # Daily returns
    returns = np.diff(prices) / prices[:-1]

    # 30-day realized volatility (annualized)
    realized_vol = float(np.std(returns[-30:]) * np.sqrt(365))

    # Latest return as proxy for daily direction
    latest_return = float(returns[-1]) if len(returns) > 0 else 0.0

    return np.array([[latest_return, realized_vol, 0.0]])  # funding_rate filled by caller
```

### Step 3 — HMM model wrapper

```python
# ats/agents/regime/hmm_model.py
import pickle, numpy as np
from pathlib import Path
from hmmlearn.hmm import GaussianHMM

MODEL_PATH = Path("models/regime_hmm.pkl")

REGIME_LABELS = {
    0: "low_vol_bull",
    1: "high_vol_bull",
    2: "choppy",
    3: "bear",
    4: "high_vol_bear",
    5: "crisis",
}

class RegimeHMM:
    def __init__(self):
        self.model: GaussianHMM | None = None
        self._load()

    def _load(self):
        if MODEL_PATH.exists():
            with open(MODEL_PATH, "rb") as f:
                self.model = pickle.load(f)

    def predict(self, features: np.ndarray) -> tuple[str, float]:
        """Returns (regime_label, confidence)."""
        if self.model is None:
            return "choppy", 0.50   # safe fallback before first training run
        state = self.model.predict(features)[-1]
        log_prob = self.model.score(features)
        confidence = min(1.0, max(0.0, (log_prob + 10) / 10))   # rough normalization
        return REGIME_LABELS[state], round(confidence, 3)

    def train(self, features: np.ndarray):
        model = GaussianHMM(n_components=6, covariance_type="full",
                            n_iter=200, random_state=42)
        model.fit(features)
        MODEL_PATH.parent.mkdir(parents=True, exist_ok=True)
        with open(MODEL_PATH, "wb") as f:
            pickle.dump(model, f)
        self.model = model

regime_hmm = RegimeHMM()
```

### Step 4 — Training script (run once before first deploy, retrain monthly)

```python
# scripts/train_regime_model.py
"""
Run: python -m scripts.train_regime_model

Downloads 2 years of daily BTC OHLCV from CoinGecko,
assembles feature matrix, trains HMM, saves to models/regime_hmm.pkl.
"""
import asyncio, httpx, numpy as np

async def fetch_btc_history(days: int = 730) -> list[dict]:
    async with httpx.AsyncClient() as client:
        resp = await client.get(
            "https://api.coingecko.com/api/v3/coins/bitcoin/market_chart",
            params={"vs_currency": "usd", "days": days, "interval": "daily"},
            timeout=30,
        )
        return resp.json().get("prices", [])

async def main():
    prices_raw = await fetch_btc_history()
    prices = np.array([p[1] for p in prices_raw])
    returns = np.diff(prices) / prices[:-1]

    # Rolling 30-day realized vol for each day
    vols = [np.std(returns[max(0, i-30):i]) * np.sqrt(365) for i in range(1, len(returns)+1)]
    vols = np.array(vols)

    # Funding rate: no historical data easily available — use 0 as placeholder for training
    funding = np.zeros(len(returns))

    features = np.column_stack([returns, vols, funding])
    features = features[~np.isnan(features).any(axis=1)]

    from ats.agents.regime.hmm_model import regime_hmm
    regime_hmm.train(features)
    print(f"Trained on {len(features)} daily bars. Model saved to models/regime_hmm.pkl")

asyncio.run(main())
```

### Step 5 — Agent 4 main loop

```python
# ats/agents/agent4_regime.py
import asyncio
from datetime import datetime, timezone
from ats.agents.regime.hmm_model import regime_hmm
from ats.agents.regime.feature_builder import build_features
from ats.agents.regime.funding_rate import get_avg_funding_rate
from ats.data.redis_client import set_json, set_float

BROADCAST_INTERVAL = 15 * 60   # 15 minutes

class Agent4RegimeDetection:
    async def run(self):
        while True:
            await self._classify_and_broadcast()
            await asyncio.sleep(BROADCAST_INTERVAL)

    async def _classify_and_broadcast(self):
        features = await build_features()
        if features is None:
            return   # not enough data yet

        funding_rate = await get_avg_funding_rate()
        features[0, 2] = funding_rate   # inject live funding rate into feature vector

        # Write raw signals to Redis so Agent 5 can read without re-fetching
        btc_vol = float(features[0, 1])
        await set_float("market:btc_vol", btc_vol)
        await set_float("market:funding_rate", funding_rate)

        label, confidence = regime_hmm.predict(features)

        await set_json("regime:current", {
            "regime": label,
            "confidence": confidence,
            "btc_vol": btc_vol,
            "funding_rate": funding_rate,
            "updated_at": datetime.now(timezone.utc).isoformat(),
        })
        await set_json("regime:updated_at",
                       {"ts": datetime.now(timezone.utc).timestamp()})
```
